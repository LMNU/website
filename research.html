  // ── Helpers ────────────────────────────────────────────────────

  const TYPE_LABELS = {

    report: 'Report',

    journalArticle: 'Journal article',

    book: 'Book',

    bookSection: 'Book chapter',

    magazineArticle: 'Magazine',

    newspaperArticle: 'Newspaper',

    podcast: 'Podcast',

    blogPost: 'Blog post',

    presentation: 'Presentation',

    conferencePaper: 'Conference paper',

  };

  // Types shown on the Research page (exclude purely outreach types)

  const RESEARCH_TYPES = ['report','journalArticle','book','bookSection','conferencePaper','presentation'];


  function typeLabel(t) {

    return TYPE_LABELS[t] || t.charAt(0).toUpperCase() + t.slice(1);

  }

  function pubHref(p) {

    return (p.url_or_download === 'download' && p.pdf) ? p.pdf : (p.url || '#');

  }

  function pubYear(p) { return p.date ? p.date.slice(0, 4) : 'n.d.'; }


  // ── State ──────────────────────────────────────────────────────

  let allPubs = [];

  let activeType = 'all';

  let activeYear = 'all';


  // ── Filtering ──────────────────────────────────────────────────

  function filtered() {

    return allPubs.filter(p => {

      if (activeType !== 'all' && p.type !== activeType) return false;

      if (activeYear !== 'all' && pubYear(p) !== activeYear) return false;

      return true;

    });

  }


  // ── Render ─────────────────────────────────────────────────────

  function render() {

    const pubs = filtered();

    const list = document.getElementById('pub-list');

    const noResults = document.getElementById('no-results');


    if (pubs.length === 0) {

      list.innerHTML = '';

      noResults.style.display = 'block';

      return;

    }

    noResults.style.display = 'none';


    list.innerHTML = pubs.map(p => `

      <a class="pub-card" href="${pubHref(p)}" target="_blank" rel="noopener">

        <div class="pub-cover-wrap">

          <img src="${p.cover_image}" alt="${p.title}" loading="lazy">

          <span class="pub-tag">${typeLabel(p.type)}</span>

          ${p.forthcoming ? '<span class="pub-forthcoming">Forthcoming</span>' : ''}

        </div>

        <p class="pub-title">${p.title}</p>

        <p class="pub-meta">${[p.source, pubYear(p)].filter(Boolean).join(' · ')}</p>

      </a>`).join('');

  }


  // ── Filter buttons ─────────────────────────────────────────────

  function buildTypeFilters(pubs) {

    const types = [...new Set(pubs.map(p => p.type))];

    const container = document.getElementById('type-filters');

    types.forEach(t => {

      const btn = document.createElement('button');

      btn.className = 'filter-btn';

      btn.dataset.type = t;

      btn.textContent = typeLabel(t);

      container.appendChild(btn);

    });

    container.addEventListener('click', e => {

      if (!e.target.matches('.filter-btn')) return;

      container.querySelectorAll('.filter-btn').forEach(b => b.classList.remove('active'));

      e.target.classList.add('active');

      activeType = e.target.dataset.type;

      render();

    });

  }


  function buildYearFilters(pubs) {

    const years = [...new Set(pubs.map(p => pubYear(p)))].sort((a, b) => b - a);

    const container = document.getElementById('year-filters');

    years.forEach(y => {

      const btn = document.createElement('button');

      btn.className = 'filter-btn';

      btn.dataset.year = y;

      btn.textContent = y;

      container.appendChild(btn);

    });

    container.addEventListener('click', e => {

      if (!e.target.matches('.filter-btn')) return;

      container.querySelectorAll('.filter-btn').forEach(b => b.classList.remove('active'));

      e.target.classList.add('active');

      activeYear = e.target.dataset.year;

      render();

    });

  }


  // ── Load ───────────────────────────────────────────────────────

  fetch('publications.json')

    .then(r => r.json())

    .then(pubs => {

      // Research page shows research-type publications

      allPubs = pubs.filter(p => RESEARCH_TYPES.includes(p.type));

      buildTypeFilters(allPubs);
